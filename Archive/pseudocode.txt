CLASS SharedData:
    // Attributes for thread-safe state management
    sensor_states = {"distance": 0, "lat": 0, "lon": 0, "sats": 0}
    update_timestamps = 0

    FUNCTION update_data(key, value):
        Set sensor_states[key] = value
        Update update_timestamps to current time

    FUNCTION get_latest_data():
        RETURN sensor_states

CLASS ProximitySensor:
    // Logic for UART-based distance sensing (13/14)
    FUNCTION get_distance():
        FLUSH UART buffer
        SEND trigger_protocol byte (0x55)
        IF 4 bytes received within timeout:
            RETURN validate_frame()

    FUNCTION validate_frame():
        // Checksum: (Header + H + L) & 0xFF
        IF header == 0xFF AND checksum_is_valid:
            distance_logic = (HighByte << 8) + LowByte
            RETURN distance_logic

CLASS CameraStreaming:
    // Logic for ESP32-S3 Camera and MJPEG stream
    FUNCTION capture_frames():
        INITIALIZE camera with init_params (pins, freq, PSRAM)
        WHILE active:
            buffer = capture image from sensor
            YIELD jpeg_encoding of buffer

    FUNCTION stream_mjpeg():
        START multipart/x-mixed-replace HTTP response

            WRITE frame to response stream

CLASS WebServer:
    // Orchestration of the Picoweb app and background tasks
    FUNCTION register_routes():
        ROUTE "/" TO WebDashboard.render_layout()
        ROUTE "/api/data" TO APIDataService.package_json()
        ROUTE "/video" TO CameraStreaming.stream_mjpeg()

    FUNCTION handle_asynchronous():
        INITIALIZE WiFi (SSID: "Vedashri")
        START background_thread for ProximitySensor and GPSInterface
        START WebServer.run()

CLASS WebDashboard:
    // Frontend logic running in the browser
    FUNCTION render_layout():
        SERVE index.html from flash memory
        LOAD map_libraries (Leaflet.js)

    FUNCTION display_stream():
        SET polling_interval = 1000ms
        EVERY interval:
            FETCH /api/data
            UPDATE map marker and text stats
            IF distance < 500: TRIGGER alert_active

MAIN:
    WebServer.handle_asynchronous()
